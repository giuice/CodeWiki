1. Add to `src/lib/merge.ts`:
```typescript
const START_MARKER = "<!-- codewiki:start -->";
const END_MARKER = "<!-- codewiki:end -->";

export function mergeMarkerSection(existing: string, newContent: string, force: boolean): string {
  const startIdx = existing.indexOf(START_MARKER);
  const endIdx = existing.indexOf(END_MARKER);
  if (startIdx !== -1 && endIdx !== -1) {
    if (!force) return existing;
    return existing.slice(0, startIdx) + START_MARKER + "\n" + newContent + "\n" + END_MARKER + existing.slice(endIdx + END_MARKER.length);
  }
  const sep = existing.length > 0 ? "\n\n" : "";
  return existing + sep + START_MARKER + "\n" + newContent + "\n" + END_MARKER + "\n";
}
```
