```typescript
import { ensureDir, writeFileSafe } from "../core/files.js";
import { scaffoldDirectories, scaffoldFiles } from "../templates/scaffold.js";
import type { SupportedTool } from "../core/types.js";
import type { ReportEntry } from "./reporter.js";

export interface ScaffoldOptions {
  root: string;
  projectName: string;
  tools: readonly SupportedTool[];
  force: boolean;
}

export async function scaffoldProject(opts: ScaffoldOptions): Promise<ReportEntry[]> {
  const report: ReportEntry[] = [];
  const dirs = scaffoldDirectories(opts.tools);
  for (const dir of dirs) {
    await ensureDir(opts.root, dir);
    report.push({ action: "created", path: dir });
  }
  const files = scaffoldFiles(opts.projectName, opts.tools);
  for (const file of files) {
    try {
      await writeFileSafe(opts.root, file.path, file.content, opts.force);
      report.push({ action: "created", path: file.path });
    } catch {
      report.push({ action: "skipped", path: file.path, reason: "exists" });
    }
  }
  return report;
}
```
