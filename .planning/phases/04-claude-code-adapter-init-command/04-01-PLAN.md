**src/lib/adapters/base.ts** — Shared helpers for all adapters. Per D-08, per-file skip/replace with continue-on-error:
```typescript
import { readFile, writeFile, chmod, mkdir } from "node:fs/promises";
import path from "node:path";
import { exists } from "../../core/files.js";
import type { ReportEntry } from "../reporter.js";

export async function copyTemplateFile(
  templatePath: string,
  targetPath: string,
  force: boolean
): Promise<ReportEntry> {
  const relPath = targetPath; // caller passes display-relative path
  const existed = await exists(targetPath);
  if (existed && !force) {
    return { action: "skipped", path: relPath, reason: "exists" };
  }
  await mkdir(path.dirname(targetPath), { recursive: true });
  const content = await readFile(templatePath, "utf8");
  await writeFile(targetPath, content, "utf8");
  return { action: existed ? "replaced" : "created", path: relPath };
}

export async function chmodExecutable(filePath: string): Promise<void> {
  await chmod(filePath, 0o755);
}

export async function copyTemplateDir(
  srcDir: string,
  destDir: string,
  force: boolean,
  displayRoot: string
): Promise<ReportEntry[]> {
  // Reads all files from srcDir recursively, copies to destDir with same relative structure
  // Returns report entries with paths relative to displayRoot
  const { readdir } = await import("node:fs/promises");
  const entries: ReportEntry[] = [];
  
  async function walk(currentSrc: string, currentDest: string): Promise<void> {
    const items = await readdir(currentSrc, { withFileTypes: true });
    for (const item of items) {
      const srcPath = path.join(currentSrc, item.name);
      const destPath = path.join(currentDest, item.name);
      if (item.isDirectory()) {
        await walk(srcPath, destPath);
      } else if (item.isFile()) {
        const displayPath = path.relative(displayRoot, destPath);
        const result = await copyTemplateFile(srcPath, destPath, force);
        entries.push({ ...result, path: displayPath });
      }
    }
  }
  await walk(srcDir, destDir);
  return entries;
}
```

const adapterFactories: Partial<Record<SupportedTool, () => Promise<ToolAdapter>>> = {
  "claude-code": async () => {
    const { ClaudeCodeAdapter } = await import("./claude.js");
    return new ClaudeCodeAdapter();
  }
};

export async function resolveAdapters(tools: SupportedTool[]): Promise<{ adapters: ToolAdapter[]; unsupported: SupportedTool[] }> {
  const adapters: ToolAdapter[] = [];
  const unsupported: SupportedTool[] = [];
  for (const tool of tools) {
    const factory = adapterFactories[tool];
    if (factory) {
      adapters.push(await factory());
    } else {
      unsupported.push(tool);
    }
  }
  return { adapters, unsupported };
}

export type { ToolAdapter, AdapterInstallOptions } from "./types.js";

export interface ReportSection {
  title: string;
  entries: ReportEntry[];
}

export function formatSectionedReport(projectName: string, sections: ReportSection[]): string {
  const lines: string[] = [`CodeWiki initialized for ${projectName}.`, ""];
  for (const section of sections) {
    if (section.entries.length === 0) continue;
    lines.push(`${section.title}:`);
    for (const entry of section.entries) {
      const detail = entry.reason ? ` (${entry.reason})` : "";
      lines.push(`  ${ACTION_SYMBOLS[entry.action]} ${entry.action.padEnd(8)} ${entry.path}${detail}`);
    }
    lines.push("");
  }
  // Summary counts across all sections
  const allEntries = sections.flatMap(s => s.entries);
  const counts = allEntries.reduce<Record<ReportAction, number>>(
    (totals, entry) => ({ ...totals, [entry.action]: totals[entry.action] + 1 }),
    { created: 0, skipped: 0, replaced: 0, failed: 0 }
  );
  const summary = ACTION_ORDER.filter(a => counts[a] > 0).map(a => `${counts[a]} ${a}`).join(", ");
  lines.push(`Summary: ${summary || "0 changes"}`);
  return lines.join("\n");
}
